Gestión del ratón

    El siguiente código, cuyo resultado puede comprobarse pinchando aquí, permite seguir la pista del ratón:

        <!DOCTYPE HTML>
        <HTML>
        <BODY>
          <script type="text/javascript">
            document.onmousemove = obtenerPosMouse;
            var pX = 0;
            var pY = 0;
            function obtenerPosMouse(e) {
              if (document.all) {
                // detecta que es un navegador IE
                pX = event.offsetX;
                pY = event.offsetY;
              } else {
                // resto de navegadores
                pX = e.pageX;
                pY = e.pageY;
              }
                // actualiza el título al cambiar la posición del ratón
              document.title = "coordenadas del ratón (x:"+pX+" y:" +pY+")";
                // actualiza barra de estado
              window.status = "coordenadas del ratón (x:"+pX+" y:" +pY+")";
            }
          </script>
        </BODY>
        </HTML>

    Internet Explorer mantiene la posición del ratón en las propiedades offsetX y offsetY, mientras que el resto de los navegadores utiliza pageX y pageY. Los valores de coordenadas son relativos a la propia ventana, a la esquina superior izquierda.

    Mensajes informativos

    alert(" Mensaje "); visualiza el mensaje y detiene el proceso hasta que el usuario pulse clic en el cuadro de diálogo. El texto que se pasa como parámetro puede incluir varias líneas utilizando como salto de línea los caracteres "\n". El título y el icono que se muestran no son personalizables.

    respuesta=confirm(" Mensaje "); es similar a alert. También detiene el proceso pero muestra dos botones y devuelve true o false dependiendo del que se pulse. También devuelve false si se cierra el cuadro de diálogo con el botón Cerrar o pulsando ESC.

    Detección del clic secundario del ratón

    En el ejemplo siguiente se detecta si el clic del ratón proviene del botón principal o del secundario. Es necesario desarrollar código diferente para IE y para los demás.

        <!DOCTYPE HTML>
        <html>
        <head>
          <title> Detección del click del ratón </title>
          <script type="text/JavaScript">
             function mouseClick1() {
                // Internet Explorer
             if ((event.button==2)||(event.button==3)) {
                alert("Clic del botón secundario.");
             } else {
                alert("Clic del botón principal.");
               }
             }
             function mouseClick2(e){
                // Mozilla Firefox
                if(e.which==3){
                  alert("Clic del botón secundario.");
                } else {
                  alert("Clic del botón principal.");
                  }
             }
                // Internet Explorer
             if (document.all) {
                document.onmousedown=mouseClick1;
             }
                // Mozilla Firefox
             if (window.sidebar) {
                document.captureEvents(Event.mousedown);
                document.onmousedown = mouseClick2;
             }
          </script>
        </head>
        Detecta si hubo clic del botón primario o secundario
        </html>

Gestión del teclado

    Dadas las diferencias existentes entre los diferentes navegadores, la gestión del teclado no suele realizarse con código JavaScript. A continuación se desarrolla un ejemplo con un cuadro de entrada de datos en el que solamente se aceptan valores numéricos, con un máximo de dos números. En primer lugar se averigua el tipo de navegador y a continuación se hace el tratamiento correspondiente:

        <!DOCTYPE HTML>
        <html>
        <form name="testform">
            Valor sólo numérico : <input type="text" name="importe">
        </form>
        <script type="text/javascript">
          document.write(verNavegador(), "<br />");
          document.forms["testform"].elements["importe"].onkeypress =  validarTeclado;
          function verNavegador() {
            // convierte a minúsculas
            var strNaveg = navigator.userAgent.toLowerCase();
 
            var strTest = strNaveg.indexOf("msie");
            if (strTest>= 0) {return "msie";};
 
            strTest = strNaveg.indexOf("netscape");
            if (strTest>= 0) {return "netscape";};
 
            strTest = strNaveg.indexOf("firefox");
            if (strTest>= 0) {return "firefox";};
 
            strTest = strNaveg.indexOf("opera");
            if (strTest>= 0) {return "opera";};
 
            return false;
          }
 
          function validarTeclado(e) {
            var strNaveg = verNavegador();
            var tecla = "";
            var testTecla = false;
            var teclasOK = "0123456789";
 
            switch (strNaveg) {
              case "msie":
                tecla = String.fromCharCode(event.keyCode);
                if (teclasOK.indexOf(tecla)>=0) {testTecla = true;};
                break;
              case "netscape":
                tecla = String.fromCharCode(e.which);
                if (teclasOK.indexOf(tecla)>=0) {testTecla = true;};
                if ((e.which == 0) && (e.keyCode> 0) || (e.which == 8)) {
                  return true;
                }
                break;
              case "firefox":
                tecla = String.fromCharCode(e.charCode);
                if (teclasOK.indexOf(tecla)>=0) {testTecla = true;};
                if ((e.charCode == 0) && (e.keyCode> 0)) {
                  return true;
                }
                break;
              default:
                // no puede validar
                return false;
            }
            if (testTecla) {
              if (document.forms["testform"].elements["importe"].value.length < 2) {
                 document.forms["testform"].elements["importe"].value+= tecla;
              }
            }
            return false; 
          }
        </script>
        </html>

    Mensaje de entrada de datos (método prompt())

    El método prompt() del objeto window permite crear una ventana de entrada de datos del usuario. Si el usuario hace click en el botón Cancelar o pulsa la tecla ESC, el método devolverá el valor nulo.

Reproducción de sonido

    HTML permite la reproducción de audio y sonido mediante la etiqueta <EMBED>, que actúa colocando un complemento del navegador en la página que sabe cómo gestionar el tipo de archivo referido en la etiqueta. Los complementos más habituales de los navegadores se utilizan para reproducir audio y sonido.

    La etiqueta <EMBED>, que tiene como principal y obligatorio el atributo src, dispone además de los siguientes:
  • src .- URL, ubicación del archivo a reproducir.
  • width .- ancho.
  • heigth .- alto.
  • align .- alineación.
  • name .- nombre.
  • pluginspage .- indica dónde obtener el software del plugin.
  • hidden .- indica si el objeto es visible o no.
  • href .- define un enlace para este objeto.
  • autostart .- reproducción automática.
  • loop .- número de repeticiones.
  • volume .- volumen.
  • starttime .- punto de inicio.
  • endtime .- punto de finalización.
    En el siguiente ejemplo el sonido se puede activar de tres modos distintos:
  • Al hacer clic en el enlace, mediante la llamada "javascript:Play()".
  • Al hacer clic en el botón, mediante el evento onclick.
  • Al pasar con el ratón sobre el segundo enlace, mediante el evento onmouseover.

        <!DOCTYPE HTML>
        <html>
          <head>
            <title>Incorporar sonidos</title>
            <script type="text/JavaScript">
              function Play() {
                eval("document.sonido.play();");
              }
            </script>
          </head>
          <body>
            <center>eventos y sonidos</center>
            <embed name='sonido' src='Todo_pasa.wav' loop="0" autostart="0" hidden="true">
            <center>
              <a href="javascript:Play()">Reprodución del sonido</a>
                <form>
                  <input type="button" value="Reproducir" onclick="Play()">
                </form>
            </center>
            <A href="javascript:void()" onmouseover="Play()">Pasar el ratón para reproducir el sonido</A>
          </body>
        </html>

    Cuyo resultado puede comprobarse pinchando aquí.
